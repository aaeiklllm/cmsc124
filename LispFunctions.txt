(defun dupli(list)
    (if (null list)
        nil
        (cons (first list)(cons (first list)(dupli(rest list))))
    )
)

(defun range(x y)
    (if (= x y)
        (append (list y))
        (if (< x y)
            (cons x(range (+ x 1) y))
            (if (> x y)
            (cons x(range (- x 1) y))
            )
        )
    )
)
